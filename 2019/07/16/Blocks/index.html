POJ 3734,又称红色病毒问题

思路：

​ 假设染色到第i个方块，红绿色都为偶数情况共有ai,红绿色恰有一个为偶数为bi,红绿色都为奇 数情况共有ci

#include

#include

#define ll long long

#define inf 1e9

#define vec vector

#define mat vector
using namespace std;
const int M=10007;

mat mul(mat& A,mat& B) {
mat C(A.size(), vec(B[0].size()));
for (int i = 0; i < A.size(); i++)
for (int k = 0; k < B[0].size(); k++)
for (int j = 0; j < B[0].size(); j++)
C[i][j] = (C[i][j] + A[i][k] * B[k][j]) % M;
return C;
}
mat pow(mat A, ll n) {
mat B(A.size(), vec(A.size()));
for (int i = 0; i < A.size(); i++)
B[i][i] = 1;
while (n > 0) {
if (n & 1)B = mul(B, A);
A = mul(A, A);
n >>= 1;
}
return B;
}
ll n;
void solve() {
mat A(3, vec(3));
A[0][0] = 2; A[0][1] = 1; A[0][2] = 0;
A[1][0] = 2; A[1][1] = 2; A[1][2] = 2;
A[2][0] = 0; A[2][1] = 1; A[2][2] = 2;
A = pow(A, n);
cout << A[0][0] << endl;
}
int main() {
int round;
cin >> round;
while (round–)
{
cin >> n;
solve();
}
return 0;
}
